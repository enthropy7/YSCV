# Criterion microbench regression guard for yscv-autograd.
# Units: microseconds (upper bound from criterion "time: [low mid high]").
#
# These are tiny (32x32 / 64x64) backward graphs dominated by graph-traversal
# and allocation overhead, so the absolute time is very sensitive to the CI
# runner's CPU. The matmul threshold was raised 15 -> 30 after a slower GitHub
# Actions VM measured ~16.7us where the code is unchanged: a 7.6us machine and
# the 6e9ebf7 baseline bench identically (verified), so this is runner-speed
# headroom, not a regression. 30us still catches any real ~2x slowdown.

autograd_backward_modes/matmul_relu_sum_32x32.max_upper_us=30
autograd_backward_modes/broadcast_add_mul_sum_64x64.max_upper_us=18
